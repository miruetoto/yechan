R함수를 파이썬에서 사용 

ro.r(''' my_fun<-function(x) x^2+1''')
my_fun_py_temp=ro.r.my_fun
def my_fun_py(a):
    rtn=r2p(my_fun_py_temp(p2r(a)))
    return rtn
my_fun_py([3,4,5])
