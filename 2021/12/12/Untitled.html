    def __init__(self, vocab_sz, n_hidden):
        self.i_h = nn.Embedding(vocab_sz,n_hidden)

    def forward(self,x):
        h= F.relu(self.h_h(self.i_h(x[:,0])))
        h= h+ self.i_h(x[:,1])
        h= F.relu(self.h_h(h))
        h= h+self.i_h(x[:,2])
        h= F.relu(self.h_h(h))
        return self.h_o(h)
