  • $W^{(0)}: 1 \times 2$
  • $W^{(1)}: 2 \times 1$

def mooyaho(input):
    return torch.sigmoid(200*input)
    
class MOOYAHO(torch.nn.Module):
    def __init__(self):
        super().__init__() # init the base class

    def forward(self, input):
        return mooyaho(input) # simply apply already implemented SiLU
