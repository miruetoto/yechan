location = [lat, lon]

zoom_start = 13

tiles = "OpenStreetMap"

import folium


width, height = 480, 350

fig = Figure(width=width, height=height)

m = folium.Map(
    location=location, tiles=tiles, width=width, height=height, zoom_start=zoom_start

fig.add_child(m)
