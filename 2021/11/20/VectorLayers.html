import folium
from folium.plugins.measure_control import MeasureControl

m = folium.Map(location=[-27.5717, -48.6256], zoom_start=9)

c = MeasureControl()

radius = 50

    location=[-27.55, -48.8],

    opacity=1,

    tooltip="I am in pixels",
).add_to(m)
