view

- a를 선언, b는 a의 view. (b는 a의 shallow copy 라고 부르기도 함)

a=np.array([[1,2],[3,4]])
b=a.view() ## 이게 shallow copy

def test(a,b):
    if id(a)==id(b): 
        print('별명')
    elif id(a)==id(b.base) or id(a.base)==id(a):
        print('뷰')
    elif id(a.base)!=id(None) and id(b.base)!=id(None) and id(a.base)==id(b.base):
        print('공통의 base를 가짐')
    else:
        print('카피이거나 아무 관련 없는 오브젝트')
