import math
pi=math.pi
n=59
ang=np.linspace(-pi,pi-2*pi/n,n)
V=np.arange(n)+1
r=1
vx=r*np.cos(ang)
vy=r*np.sin(ang)
f1=vx*0

f2=vx*0
f2[vy<0]=3+np.random.normal(size=sum(vy<0),scale=0.1)
