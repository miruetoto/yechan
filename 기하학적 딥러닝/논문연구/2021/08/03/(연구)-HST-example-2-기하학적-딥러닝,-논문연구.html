np.random.seed(777)
import math
pi=math.pi
n=59
ang=np.linspace(-pi,pi-2*pi/n,n)
V=np.arange(n)+1
r=1
vx=r*np.cos(ang)
vy=r*np.sin(ang)
f1=vx*0

f2=vx*0
f2[vy<0]=3+np.random.normal(size=sum(vy<0),scale=0.1)
f2[vy>0]= -3+np.random.normal(size=sum(vy>0),scale=0.1)

Σ=l2distance(np.matrix([vx,vy]).T)
θ=0.05
W=np.exp(-Σ**2/(2*θ**2))
E=W>0
plt.plot(W[0,:].T)
