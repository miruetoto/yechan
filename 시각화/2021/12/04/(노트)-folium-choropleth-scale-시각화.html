import pandas as pd
import folium

import json
import requests

global_dict = json.loads(requests.get(global_distriction_jsonurl).text)

df=pd.read_csv('https://raw.githubusercontent.com/guebin/2021DV/master/_notebooks/2021-11-22-prov.csv')

linear = cmp.LinearColormap(
    ['white', 'blue'],
    vmin=0, vmax=13549577,
    caption='Color Scale for Map' #Caption for Color scale or Legend
)
linear.add_to(m)

folium.GeoJson(
    global_dict,
    style_function=lambda feature: {
        'fillColor': linear(_dict[feature['properties']['name']]),
        'color': 'black',     
        'weight': 1,
        'fillOpacity': 0.8,
        'dashArray': '5, 3'  
    }
).add_to(m)
